# QuickCheck 并发互斥锁 Implementation Plan

**Goal:** 给 scripts/quickcheck.sh 加互斥锁，同一 repo 同时只能跑一个 vitest 实例

**Architecture:** flock（Linux）+ mkdir 原子锁（macOS fallback）+ trap 清理。锁超时 2s 拿不到则跳过（exit 0，不 block push）

**Tech:** bash 4+, macOS/Linux 兼容

```typescript
import { describe, it, expect, beforeEach, afterEach } from 'vitest';
import { execSync, spawn } from 'child_process';
import { mkdtempSync, rmSync, writeFileSync, mkdirSync, existsSync } from 'fs';
import { tmpdir } from 'os';
import { join } from 'path';

const SCRIPT = join(process.cwd(), '..', '..', 'scripts', 'quickcheck.sh');

describe('quickcheck.sh — 并发互斥锁', () => {
  let fakeRepo: string;
  let fakeScript: string;

  beforeEach(() => {
    // 构造 fake repo + fake quickcheck（只 sleep 3s 测锁行为，不跑真 vitest）
    fakeRepo = mkdtempSync(join(tmpdir(), 'qcmutex-'));
    execSync(`git init -q "${fakeRepo}"`, { stdio: 'pipe' });
    fakeScript = join(fakeRepo, 'quickcheck.sh');
    // 测试版 quickcheck — 加锁 + sleep 2s + 写标记文件
    // 注意：这里只复制锁逻辑部分，其他省略
    const real = require('fs').readFileSync(SCRIPT, 'utf8');
    // 替换实际工作为 sleep + touch marker
    const testVersion = real.replace(
      /# === 以下为实际检查逻辑[\s\S]*$/,
      '\nsleep 2\ntouch "${REPO_ROOT}/ran.$$"\nexit 0\n'
    );
    writeFileSync(fakeScript, testVersion);
    execSync(`chmod +x "${fakeScript}"`);
  });

  afterEach(() => rmSync(fakeRepo, { recursive: true, force: true }));

  it('两次并发调用，只有一个跑完，另一个跳过', async () => {
    const run1 = spawn('bash', [fakeScript], { cwd: fakeRepo, stdio: 'pipe' });
    // 50ms 后启动第二个，确保第一个已拿到锁
    await new Promise(r => setTimeout(r, 50));
    const run2Output = execSync(`bash "${fakeScript}" 2>&1`, { cwd: fakeRepo, encoding: 'utf8' });

    // 第二个应跳过
    expect(run2Output).toMatch(/跳过|skip|已在运行|another quickcheck/i);

    // 等第一个跑完
    await new Promise<void>(resolve => run1.on('close', () => resolve()));

    // 只有一个 marker 文件
    const markers = require('fs').readdirSync(fakeRepo).filter((f: string) => f.startsWith('ran.'));
    expect(markers.length).toBe(1);
  }, 10000);

  it('锁在脚本结束后自动释放（下一次能正常跑）', async () => {
    // 先跑一次，等完
    execSync(`bash "${fakeScript}"`, { cwd: fakeRepo, encoding: 'utf8' });
    expect(existsSync(join(fakeRepo, '.git/quickcheck.lock')) || existsSync(join(fakeRepo, '.git/quickcheck.lockdir'))).toBe(false);

    // 再跑一次，应成功（不跳过）
    const out = execSync(`bash "${fakeScript}"`, { cwd: fakeRepo, encoding: 'utf8' });
    expect(out).not.toMatch(/跳过|skip/i);
  }, 10000);
});
```

## Task 2: 在 quickcheck.sh 加互斥锁

**Files:** `scripts/quickcheck.sh`（修改前 10 行）

- [ ] Step 1: 在 `cd "$REPO_ROOT"` 之后、worktree 软链之前，插入锁逻辑

```bash
# ============================================================================
# 互斥锁：同一 repo 最多一个 quickcheck 实例
# 支持 flock (Linux) + mkdir 原子锁 fallback (macOS)
# ============================================================================
_GIT_DIR="$(git rev-parse --git-dir 2>/dev/null || echo "$REPO_ROOT/.git")"
_LOCK_FILE="${_GIT_DIR}/quickcheck.lock"
_LOCK_DIR="${_GIT_DIR}/quickcheck.lockdir"

if command -v flock &>/dev/null; then
    exec 200>"$_LOCK_FILE"
    if ! flock -w 2 -n 200; then
        echo "⏭️  另一个 quickcheck 正在运行，跳过本次预检" >&2
        exit 0
    fi
    # flock 在 fd 200 上，exit 时自动释放
else
    # macOS fallback: mkdir 原子锁
    _lock_try=0
    until mkdir "$_LOCK_DIR" 2>/dev/null; do
        _lock_try=$((_lock_try + 1))
        if [[ $_lock_try -ge 20 ]]; then
            echo "⏭️  另一个 quickcheck 正在运行（mkdir 锁 2s 超时），跳过本次预检" >&2
            exit 0
        fi
        sleep 0.1
    done
    # 清理 trap — 无论正常退出还是被 kill 都释放锁
    trap 'rmdir "$_LOCK_DIR" 2>/dev/null || true' EXIT INT TERM
fi
```
